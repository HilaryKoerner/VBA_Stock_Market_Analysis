' StarCounter
range(i2).value=Total

dim name=Total.double

dim StarsCounter as integer

    'Loop through each row (rows are students)
    for i = 2 to 51
    StarsCounter = 0

        'loop through each column (columns are star ratings)
        for j = 4 to 8

            'add each column together

            'if a column contains the word "Full-Star" then count 1
            If Cells(i, j).Value = "Full-Star" Then
            
                'Add 1 to the star counter
                StarCounter = StarCounter + 1
            
            end if
          'Show the number of stars rated per students
          cells(i,9).value = StarCounter
        Next j
        
    Next i

end sub
